1   package org.infinispan.topology;
2   
3   import org.infinispan.statetransfer.StateTransferManager;
4   
5   /**
6    * The link between {@link LocalTopologyManager} and {@link StateTransferManager}.
7    *
8    * @author Dan Berindei
9    * @since 5.2
10   */
11  public interface CacheTopologyHandler {
12  
13     /**
14      * Invoked when the CH has to be immediately updated because of a leave or when the state transfer has completed
15      * and we have to install a permanent CH (pendingCH == null). A state transfer is not always required.
16      */
17     void updateConsistentHash(CacheTopology cacheTopology);
18  
19     /**
20      * Invoked when state transfer has to be started.
21      *
22      * The caller will not consider the local rebalance done when this method returns. Instead, the handler
23      * will have to call {@link LocalTopologyManager#confirmRebalance(String, int, int, Throwable)}
24      */
25     void rebalance(CacheTopology cacheTopology);
26  }
