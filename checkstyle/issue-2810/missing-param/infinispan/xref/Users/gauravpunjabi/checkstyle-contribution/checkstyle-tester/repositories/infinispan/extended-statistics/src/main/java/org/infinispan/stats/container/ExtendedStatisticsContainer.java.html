1   package org.infinispan.stats.container;
2   
3   import org.infinispan.stats.ExtendedStatisticNotFoundException;
4   
5   /**
6    * Contains the statistic's values and allows to perform modifications on them.
7    *
8    * @author Pedro Ruivo
9    * @since 6.0
10   */
11  public interface ExtendedStatisticsContainer {
12  
13     /**
14      * it adds the value to the statistic. If the statistic does not exist in this container, it fails silently
15      */
16     void addValue(ExtendedStatistic statistic, double value);
17  
18     /**
19      * @return the current value of the statistic
20      * @throws ExtendedStatisticNotFoundException
21      *          if the statistic was not found in this container
22      */
23     double getValue(ExtendedStatistic statistic) throws ExtendedStatisticNotFoundException;
24  
25     /**
26      * it merges in {@code this} the statistic's values in {@code other}. If for some reason the {@code other} cannot be
27      * merged, it fails silently
28      */
29     void mergeTo(ConcurrentGlobalContainer globalContainer);
30  }
