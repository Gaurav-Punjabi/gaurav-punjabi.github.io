1   package org.infinispan.server.hotrod.configuration;

3   import org.infinispan.configuration.cache.LockingConfigurationBuilder;
4   import org.infinispan.configuration.cache.StateTransferConfigurationBuilder;
5   import org.infinispan.configuration.cache.SyncConfigurationBuilder;
6   
7   /**
8    * HotRodServerChildConfigurationBuilder.
9    *
10   * @author Tristan Tarrant
11   * @since 7.0
12   */
13  public interface HotRodServerChildConfigurationBuilder {
14  
15     AuthenticationConfigurationBuilder authentication();
16  
17     /**
18      * Sets the external address of this node, i.e. the address which clients will connect to
19      */
20     HotRodServerChildConfigurationBuilder proxyHost(String proxyHost);
21  
22     /**
23      * Sets the external port of this node, i.e. the port which clients will connect to
24      */
25     HotRodServerChildConfigurationBuilder proxyPort(int proxyPort);
26  
27     /**
28      * Configures the lock acquisition timeout for the topology cache. See {@link LockingConfigurationBuilder#lockAcquisitionTimeout(long)}. Defaults to 10 seconds
29      */
30     HotRodServerChildConfigurationBuilder topologyLockTimeout(long topologyLockTimeout);
31  
32     /**
33      * Configures the replication timeout for the topology cache. See {@link SyncConfigurationBuilder#replTimeout(long)}. Defaults to 10 seconds
34      */
35     HotRodServerChildConfigurationBuilder topologyReplTimeout(long topologyReplTimeout);
36  
37     /**
38      * Configures whether to enable waiting for initial state transfer for the topology cache. See {@link StateTransferConfigurationBuilder#awaitInitialTransfer(boolean)}
39      */
40     HotRodServerChildConfigurationBuilder topologyAwaitInitialTransfer(boolean topologyAwaitInitialTransfer);
41  
42     /**
43      * Configures whether to enable state transfer for the topology cache. If disabled, a {@link org.infinispan.persistence.cluster.ClusterLoader} will be used to lazily retrieve topology information from the other nodes.
44      * Defaults to true.
45      */
46     HotRodServerChildConfigurationBuilder topologyStateTransfer(boolean topologyStateTransfer);
47  
48  }
