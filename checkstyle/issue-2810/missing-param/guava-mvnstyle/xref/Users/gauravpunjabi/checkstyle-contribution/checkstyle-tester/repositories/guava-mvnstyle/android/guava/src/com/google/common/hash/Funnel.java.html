15  package com.google.common.hash;
16  
17  import com.google.common.annotations.Beta;
18  import com.google.errorprone.annotations.DoNotMock;
19  import java.io.Serializable;
20  
21  /**
22   * An object which can send data from an object of type {@code T} into a {@code PrimitiveSink}.
23   * Implementations for common types can be found in {@link Funnels}.
24   *
25   * <p>Note that serialization of {@linkplain BloomFilter bloom filters} requires the proper
26   * serialization of funnels. When possible, it is recommended that funnels be implemented as a
27   * single-element enum to maintain serialization guarantees. See Effective Java (2nd Edition), Item
28   * 3: "Enforce the singleton property with a private constructor or an enum type". For example:
29   *
30   * <pre>{@code
31   * public enum PersonFunnel implements Funnel<Person> {
32   *   INSTANCE;
33   *   public void funnel(Person person, PrimitiveSink into) {
34   *     into.putUnencodedChars(person.getFirstName())
35   *         .putUnencodedChars(person.getLastName())
36   *         .putInt(person.getAge());
37   *   }
38   * }
39   * }</pre>
40   *
41   * @author Dimitris Andreou
42   * @since 11.0
43   */
44  @Beta
45  @DoNotMock("Implement with a lambda")
46  public interface Funnel<T> extends Serializable {
47  
48    /**
49     * Sends a stream of data from the {@code from} object into the sink {@code into}. There is no
50     * requirement that this data be complete enough to fully reconstitute the object later.
51     *
52     * @since 12.0 (in Guava 11.0, {@code PrimitiveSink} was named {@code Sink})
53     */
54    void funnel(T from, PrimitiveSink into);
55  }
