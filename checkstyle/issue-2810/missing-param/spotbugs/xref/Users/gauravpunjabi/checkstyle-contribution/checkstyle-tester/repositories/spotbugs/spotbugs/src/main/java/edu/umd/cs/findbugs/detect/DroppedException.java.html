88      boolean isChecked(String c) {
89          if (!causes.add(c)) {
90              return checkedCauses.contains(c);
91          }
92          try {
93              if (Hierarchy.isSubtype(c, "java.lang.Exception") && !Hierarchy.isSubtype(c, "java.lang.RuntimeException")) {
94                  checkedCauses.add(c);
95                  return true;
96              }
97          } catch (ClassNotFoundException e) {
98              bugReporter.reportMissingClass(e);
99          }
100         return false;
101     }

103     private int getUnsignedShort(byte[] a, int i) {
104         return asUnsignedByte(a[i]) << 8 | asUnsignedByte(a[i + 1]);
105     }
106 
107     @edu.umd.cs.findbugs.annotations.SuppressFBWarnings(value = "UC_USELESS_CONDITION", justification = "To be fixed in SpotBugs 4.0.0, see https://github.com/spotbugs/spotbugs/issues/84")
108     @Override
109     public void visit(Code obj) {
110 
111         CodeException[] exp = obj.getExceptionTable();
112         LineNumberTable lineNumbers = obj.getLineNumberTable();
113         if (exp == null) {
114             return;
115         }
116         byte[] code = obj.getCode();
117 
118         for (CodeException aExp : exp) {
119             int handled = aExp.getHandlerPC();
120             int start = aExp.getStartPC();
121             int end = aExp.getEndPC();
122             int cause = aExp.getCatchType();
123             boolean exitInTryBlock = false;
124             if (DEBUG) {
125                 System.out.println("start = " + start + ", end = " + end + ", codeLength = " + code.length + ", handled = "
126                         + handled);
127             }
128 
129             for (int j = start; j <= end && j < code.length;) {
130                 int opcode = asUnsignedByte(code[j]);
131                 if (Const.getNoOfOperands(opcode) < 0) {
132                     exitInTryBlock = true;
133                     break;
134                 }
135                 j += 1 + Const.getNoOfOperands(opcode);
136                 if (opcode >= Const.IRETURN && opcode <= Const.RETURN || opcode >= Const.IFEQ && opcode <= Const.GOTO && (opcode != Const.GOTO || j < end)) {
137                     exitInTryBlock = true;
138                     if (DEBUG) {
139                         System.out.println("\texit: " + opcode + " in " + getFullyQualifiedMethodName());
140                     }
141                     break;
142                 }
143 
144             }
145 
146             if (exitInTryBlock) {
147                 if (DEBUG) {
148                     System.out.println("Exit in try block");
149                 }
150                 continue;
151             }
152             if (handled < 5) {
153                 continue;
154             }
155             @DottedClassName String causeName;
156             if (cause == 0) {
157                 causeName = "java.lang.Throwable";
158             } else {
159                 causeName = Utility.compactClassName(getConstantPool().getConstantString(cause, Const.CONSTANT_Class), false);
160                 if (!isChecked(causeName)) {
161                     continue;
162                 }
163             }
164 
165             int jumpAtEnd = 0;
166             if (end < code.length && asUnsignedByte(code[end]) == Const.GOTO) {
167                 jumpAtEnd = getUnsignedShort(code, end + 1);
168                 if (jumpAtEnd < handled) {
169                     jumpAtEnd = 0;
170                 }
171             }
172 
173             int opcode = asUnsignedByte(code[handled]);
174             int afterHandler = 0;
175             if (DEBUG) {
176                 System.out.println("DE:\topcode is " + Const.getOpcodeName(opcode) + ", " + asUnsignedByte(code[handled + 1]));
177             }
178             boolean drops = false;
179             boolean startsWithASTORE03 = opcode >= Const.ASTORE_0 && opcode <= Const.ASTORE_3;
180             if (startsWithASTORE03 && asUnsignedByte(code[handled + 1]) == Const.RETURN) {
181                 if (DEBUG) {
182                     System.out.println("Drop 1");
183                 }
184                 drops = true;
185                 afterHandler = handled + 1;
186             }
187             if (handled + 2 < code.length && opcode == Const.ASTORE && asUnsignedByte(code[handled + 2]) == Const.RETURN) {
188                 drops = true;
189                 afterHandler = handled + 2;
190                 if (DEBUG) {
191                     System.out.println("Drop 2");
192                 }
193             }
194             if (handled + 3 < code.length && !exitInTryBlock) {
195                 if (DEBUG) {
196                     System.out.println("DE: checking for jumps");
197                 }
198                 if (startsWithASTORE03 && asUnsignedByte(code[handled - 3]) == Const.GOTO) {
199                     int offsetBefore = getUnsignedShort(code, handled - 2);
200                     if (DEBUG) {
201                         System.out.println("offset before = " + offsetBefore);
202                     }
203                     if (offsetBefore == 4) {
204                         drops = true;
205                         afterHandler = handled + 1;
206                         if (DEBUG) {
207                             System.out.println("Drop 3");
208                         }
209                     }
210                 }
211                 if (opcode == Const.ASTORE && asUnsignedByte(code[handled - 3]) == Const.GOTO) {
212                     int offsetBefore = getUnsignedShort(code, handled - 2);
213                     if (offsetBefore == 5) {
214                         drops = true;
215                         afterHandler = handled + 2;
216                         if (DEBUG) {
217                             System.out.println("Drop 4");
218                         }
219                     }
220                 }
221                 if (startsWithASTORE03 && asUnsignedByte(code[handled + 1]) == Const.GOTO && asUnsignedByte(code[handled - 3]) == Const.GOTO) {
222                     int offsetBefore = getUnsignedShort(code, handled - 2);
223                     int offsetAfter = getUnsignedShort(code, handled + 2);
224 
225                     if (offsetAfter > 0 && offsetAfter + 4 == offsetBefore) {
226                         drops = true;
227                         afterHandler = handled + 4;
228                         if (DEBUG) {
229                             System.out.println("Drop 5");
230                         }
231                     }
232                 }
233 
234                 if (opcode == Const.ASTORE && asUnsignedByte(code[handled + 2]) == Const.GOTO && asUnsignedByte(code[handled - 3]) == Const.GOTO) {
235                     int offsetBefore = getUnsignedShort(code, handled - 2);
236                     int offsetAfter = getUnsignedShort(code, handled + 3);
237 
238                     if (offsetAfter > 0 && offsetAfter + 5 == offsetBefore) {
239                         drops = true;
240                         afterHandler = handled + 5;
241                         if (DEBUG) {
242                             System.out.println("Drop 6");
243                         }
244                     }
245                 }
246 
247             }
248 
249             boolean multiLineHandler = false;
250             if (DEBUG) {
251                 System.out.println("afterHandler = " + afterHandler + ", handled = " + handled);
252             }
253             if (afterHandler > handled && lineNumbers != null) {
254                 int startHandlerLinenumber = lineNumbers.getSourceLine(handled);
255 
256                 int endHandlerLinenumber = getNextExecutableLineNumber(lineNumbers, afterHandler) - 1;
257                 if (DEBUG) {
258                     System.out.println("Handler in lines " + startHandlerLinenumber + "-" + endHandlerLinenumber);
259                 }
260                 if (endHandlerLinenumber > startHandlerLinenumber) {
261                     multiLineHandler = true;
262                     if (DEBUG) {
263                         System.out.println("Multiline handler");
264                     }
265                 }
266             }
267 
268             if (end - start >= 4 && drops && !"java.lang.InterruptedException".equals(causeName)
269                     && !"java.lang.CloneNotSupportedException".equals(causeName)) {
270                 int priority = NORMAL_PRIORITY;
271                 if (exitInTryBlock) {
272                     priority++;
273                 }
274                 if (end - start == 4) {
275                     priority++;
276                 }
277                 SourceLineAnnotation srcLine = SourceLineAnnotation.fromVisitedInstruction(this.classContext, this, handled);
278                 if (srcLine != null && LOOK_IN_SOURCE_TO_FIND_COMMENTED_CATCH_BLOCKS) {
279                     if (catchBlockHasComment(srcLine)) {
280                         return;
281                     } else {
282                         priority++;
283                     }
284                 } else {
285                     // can't look at source
286                     if (lineNumbers == null || multiLineHandler) {
287                         priority += 2;
288                     }
289                 }
290                 if ("java.lang.Error".equals(causeName) || "java.lang.Exception".equals(causeName) || "java.lang.Throwable".equals(causeName)
291                         || "java.lang.RuntimeException".equals(causeName)) {
292                     priority--;
293                     if (end - start > 30) {
294                         priority--;
295                     }
296                 }
297 
298                 int register = -1;
299                 if (startsWithASTORE03) {
300                     register = opcode - Const.ASTORE_0;
301                 } else if (opcode == Const.ASTORE) {
302                     register =  asUnsignedByte(code[handled + 1]);
303                 }
304                 if (register >= 0) {
305                     LocalVariableAnnotation lva = LocalVariableAnnotation.getLocalVariableAnnotation(getMethod(), register, handled+2, handled+1);
306                     String name = lva.getName();
307                     if (DEBUG) {
308                         System.out.println("Name: " + name);
309                     }
310                     if (name.startsWith("ignore") || name.startsWith("cant")) {
311                         continue;
312                     }
313                 }
314 
315 
316                 if (DEBUG) {
317                     System.out.println("Priority is " + priority);
318                 }
319                 if (priority > LOW_PRIORITY) {
320                     return;
321                 }
322                 if (priority < HIGH_PRIORITY) {
323                     priority = HIGH_PRIORITY;
324                 }
325                 if (DEBUG) {
326                     System.out.println("reporting warning");
327                 }
328 
329                 BugInstance bugInstance = new BugInstance(this, exitInTryBlock ? "DE_MIGHT_DROP" : "DE_MIGHT_IGNORE", priority)
330                 .addClassAndMethod(this);
331                 bugInstance.addClass(causeName).describe("CLASS_EXCEPTION");
332                 bugInstance.addSourceLine(srcLine);
333                 bugAccumulator.accumulateBug(bugInstance, srcLine);
334 
335             }
336         }
337     }

339     private int getNextExecutableLineNumber(LineNumberTable linenumbers, int PC) {
340         LineNumber[] entries = linenumbers.getLineNumberTable();
341         int beforePC = 0;
342         int i = 0;
343         for (; i < entries.length && entries[i].getStartPC() < PC; i++) {
344             int line = entries[i].getLineNumber();
345             if (line > beforePC) {
346                 beforePC = line;
347             }
348         }
349 
350         if (i < entries.length) {
351             int secondChoice = entries[i].getLineNumber();
352             for (; i < entries.length; i++) {
353                 int line = entries[i].getLineNumber();
354                 if (line > beforePC) {
355                     return line;
356                 }
357             }
358             return secondChoice;
359         } else {
360             return entries[entries.length - 1].getLineNumber();
361         }
362     }

364     private static final int START = 0;
365 
366     private static final int CATCH = 1;
367 
368     private static final int OPEN_PAREN = 2;
369 
370     private static final int CLOSE_PAREN = 3;
371 
372     private static final int OPEN_BRACE = 4;
373 
374     /**
375      * Maximum number of lines we look backwards to find the "catch" keyword.
376      * Looking backwards is necessary when the indentation style puts the open
377      * brace on a different line from the catch clause.
378      */
379     private static final int NUM_CONTEXT_LINES = 3;
380 
381     /**
382      * The number of lines that we'll scan to look at the source for a catch
383      * block.
384      */
385     private static final int MAX_LINES = 7;
386 
387     /**
388      * Analyze a class's source code to see if there is a comment (or other
389      * text) in a catch block we have marked as dropping an exception.
390      *
391      * @return true if there is a comment in the catch block, false if not (or
392      *         if we can't tell)
393      */
394     private boolean catchBlockHasComment(SourceLineAnnotation srcLine) {
395         if (!LOOK_IN_SOURCE_TO_FIND_COMMENTED_CATCH_BLOCKS) {
396             return false;
397         }
398 
399         SourceFinder sourceFinder = AnalysisContext.currentAnalysisContext().getSourceFinder();
400         try {
401             SourceFile sourceFile = sourceFinder.findSourceFile(srcLine.getPackageName(), srcLine.getSourceFile());
402             int startLine = srcLine.getStartLine();
403 
404             int scanStartLine = startLine - NUM_CONTEXT_LINES;
405             if (scanStartLine < 1) {
406                 scanStartLine = 1;
407             }
408 
409             int offset = sourceFile.getLineOffset(scanStartLine - 1);
410             if (offset < 0)
411             {
412                 return false; // Source file has changed?
413             }
414             Tokenizer tokenizer = new Tokenizer(UTF8.reader(sourceFile.getInputStreamFromOffset(offset)));
415 
416             // Read the tokens into an ArrayList,
417             // keeping track of where the catch block is reported
418             // to start
419             ArrayList<Token> tokenList = new ArrayList<>(40);
420             int eolOfCatchBlockStart = -1;
421             for (int line = scanStartLine; line < scanStartLine + MAX_LINES;) {
422                 Token token = tokenizer.next();
423                 int kind = token.getKind();
424                 if (kind == Token.EOF) {
425                     break;
426                 }
427 
428                 if (kind == Token.EOL) {
429                     if (line == startLine) {
430                         eolOfCatchBlockStart = tokenList.size();
431                     }
432                     ++line;
433                 }
434 
435                 tokenList.add(token);
436             }
437 
438             if (eolOfCatchBlockStart < 0)
439             {
440                 return false; // Couldn't scan line reported as start of catch
441                 // block
442             }
443 
444             // Starting at the end of the line reported as the start of the
445             // catch block,
446             // scan backwards for the token "catch".
447             ListIterator<Token> iter = tokenList.listIterator(eolOfCatchBlockStart);
448             boolean foundCatch = false;
449 
450             while (iter.hasPrevious()) {
451                 Token token = iter.previous();
452                 if (token.getKind() == Token.WORD && "catch".equals(token.getLexeme())) {
453                     foundCatch = true;
454                     break;
455                 }
456             }
457 
458             if (!foundCatch)
459             {
460                 return false; // Couldn't find "catch" keyword
461             }
462 
463             // Scan forward from the "catch" keyword to see what text
464             // is in the handler block. If the block is non-empty,
465             // then we suppress the warning (on the theory that the
466             // programmer has indicated that there is a good reason
467             // that the exception is ignored).
468             boolean done = false;
469             int numLines = 0;
470             int state = START;
471             int level = 0;
472             do {
473                 if (!iter.hasNext()) {
474                     break;
475                 }
476 
477                 Token token = iter.next();
478                 int type = token.getKind();
479                 String value = token.getLexeme();
480 
481                 switch (type) {
482                 case Token.EOL:
483                     if (DEBUG) {
484                         System.out.println("Saw token: [EOL]");
485                     }
486                     ++numLines;
487                     if (numLines >= MAX_LINES) {
488                         done = true;
489                     }
490                     break;
491                 default:
492                     if (DEBUG) {
493                         System.out.println("Got token: " + value);
494                     }
495                     switch (state) {
496                     case START:
497                         if ("catch".equals(value)) {
498                             state = CATCH;
499                         }
500                         break;
501                     case CATCH:
502                         if ("(".equals(value)) {
503                             state = OPEN_PAREN;
504                         }
505                         break;
506                     case OPEN_PAREN:
507                         if (")".equals(value)) {
508                             if (level == 0) {
509                                 state = CLOSE_PAREN;
510                             } else {
511                                 --level;
512                             }
513                         } else if ("(".equals(value)) {
514                             ++level;
515                         }
516                         break;
517                     case CLOSE_PAREN:
518                         if ("{".equals(value)) {
519                             state = OPEN_BRACE;
520                         }
521                         break;
522                     case OPEN_BRACE:
523                         boolean closeBrace = "}".equals(value);
524                         if (DEBUG && !closeBrace) {
525                             System.out.println("Found a comment in catch block: " + value);
526                         }
527                         return !closeBrace;
528                     }
529                     break;
530                 }
531             } while (!done);
532         } catch (IOException e) {
533             // Ignored; we'll just assume there is no comment
534 
535         }
536         return false;
537     }
