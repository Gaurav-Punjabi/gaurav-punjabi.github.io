22  package edu.umd.cs.findbugs.detect;

24  import org.apache.bcel.Const;
25  import org.apache.bcel.classfile.Code;
26  import org.apache.bcel.classfile.JavaClass;
27  import org.apache.bcel.classfile.Method;

29  import edu.umd.cs.findbugs.BugInstance;
30  import edu.umd.cs.findbugs.BugReporter;
31  import edu.umd.cs.findbugs.BytecodeScanningDetector;
32  import edu.umd.cs.findbugs.Lookup;
33  import edu.umd.cs.findbugs.MethodAnnotation;
34  import edu.umd.cs.findbugs.ba.AnalysisContext;
35  import edu.umd.cs.findbugs.ba.ClassContext;
36  import edu.umd.cs.findbugs.ba.XClass;
37  import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;
38  import edu.umd.cs.findbugs.classfile.ClassDescriptor;
39  import edu.umd.cs.findbugs.classfile.Global;
40  
41  public class InvalidJUnitTest extends BytecodeScanningDetector {
42  
43      private static final int SEEN_NOTHING = 0;
44  
45      private static final int SEEN_ALOAD_0 = 1;
46  
47      private final BugReporter bugReporter;
48  
49      private int state;
50  
51      public InvalidJUnitTest(BugReporter bugReporter) {
52          this.bugReporter = bugReporter;
53      }
54  
55      boolean directChildOfTestCase;
56  
57      @Override
58      public void visitClassContext(ClassContext classContext) {
59          if (!enabled()) {
60              return;
61          }
62  
63          JavaClass jClass = classContext.getJavaClass();
64          XClass xClass = classContext.getXClass();
65  
66          try {
67  
68              if (!isJunit3TestCase(xClass)) {
69                  return;
70              }
71              if ((jClass.getAccessFlags() & Const.ACC_ABSTRACT) == 0) {
72                  if (!hasTestMethods(jClass)) {
73                      bugReporter.reportBug(new BugInstance(this, "IJU_NO_TESTS", LOW_PRIORITY).addClass(jClass));
74                  }
75              }
76              directChildOfTestCase = "junit.framework.TestCase".equals(jClass.getSuperclassName());
77              jClass.accept(this);
78          } catch (ClassNotFoundException cnfe) {
79              bugReporter.reportMissingClass(cnfe);
80          }
81  
82      }
83  
84      private boolean isJunit3TestCase(XClass jClass) throws ClassNotFoundException {
85          ClassDescriptor sDesc = jClass.getSuperclassDescriptor();
86          if (sDesc == null) {
87              return false;
88          }
89          String sName = sDesc.getClassName();
90          if (sName.equals("junit/framework/TestCase")) {
91              return true;
92          }
93          if (sName.equals("java/lang/Object")) {
94              return false;
95          }
96  
97          try {
98              XClass sClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, sDesc);
99              if (sClass == null) {
100                 return false;
101             }
102             return isJunit3TestCase(sClass);
103         } catch (CheckedAnalysisException e) {
104             return false;
105         }
106 
107     }
108 
109     private boolean hasTestMethods(JavaClass jClass) {
110         boolean foundTest = false;
111         Method[] methods = jClass.getMethods();
112         for (Method m : methods) {
113             if (m.isPublic() && m.getName().startsWith("test") && m.getSignature().equals("()V")) {
114                 return true;
115             }
116             if (m.getName().startsWith("runTest") && m.getSignature().endsWith("()V")) {
117                 return true;
118             }
119         }
120         if (hasSuite(methods)) {
121             return true;
122         }
123 
124         try {
125             JavaClass sClass = jClass.getSuperClass();
126             if (sClass != null) {
127                 return hasTestMethods(sClass);
128             }
129         } catch (ClassNotFoundException e) {
130             AnalysisContext.reportMissingClass(e);
131         }
132 
133         return false;
134     }
135 
136     /** is there a JUnit3TestSuite */
137     private boolean hasSuite(Method[] methods) {
138         for (Method m : methods) {
139             if (m.getName().equals("suite") && m.isPublic() && m.isStatic()
140                     // && m.getReturnType().equals(junit.framework.Test.class)
141                     // && m.getArgumentTypes().length == 0
142                     && m.getSignature().equals("()Ljunit/framework/Test;")) {
143                 return true;
144             }
145         }
146         return false;
147     }
148 
149     /**
150      * Check whether or not this detector should be enabled. The detector is
151      * disabled if the TestCase class cannot be found (meaning we don't have
152      * junit.jar on the aux classpath).
153      *
154      * @return true if it should be enabled, false if not
155      */
156     private boolean enabled() {
157         return true;
158     }
159 
160     @Override
161     public void visit(Method obj) {
162         if (getMethodName().equals("suite") && !obj.isStatic()) {
163             bugReporter.reportBug(new BugInstance(this, "IJU_SUITE_NOT_STATIC", NORMAL_PRIORITY).addClassAndMethod(this));
164         }
165 
166         if (getMethodName().equals("suite") && obj.getSignature().startsWith("()") && obj.isStatic()) {
167             if ((!obj.getSignature().equals("()Ljunit/framework/Test;") && !obj.getSignature().equals(
168                     "()Ljunit/framework/TestSuite;"))
169                     || !obj.isPublic()) {
170                 bugReporter.reportBug(new BugInstance(this, "IJU_BAD_SUITE_METHOD", NORMAL_PRIORITY).addClassAndMethod(this));
171             }
172 
173         }
174     }
175 
176     private boolean sawSuperCall;
177 
178     @Override
179     public void visit(Code obj) {
180         if (!directChildOfTestCase && (getMethodName().equals("setUp") || getMethodName().equals("tearDown"))
181                 && !getMethod().isPrivate() && getMethodSig().equals("()V")) {
182             sawSuperCall = false;
183             super.visit(obj);
184             if (sawSuperCall) {
185                 return;
186             }
187             JavaClass we = Lookup.findSuperImplementor(getThisClass(), getMethodName(), "()V", bugReporter);
188             if (we != null && !we.getClassName().equals("junit.framework.TestCase")) {
189                 // OK, got a bug
190                 int offset = 0;
191                 if (getMethodName().equals("tearDown")) {
192                     offset = obj.getCode().length - 1;
193                 }
194                 Method superMethod = Lookup.findImplementation(we, getMethodName(), "()V");
195                 Code superCode = superMethod.getCode();
196                 if (superCode != null && superCode.getCode().length > 3) {
197                     bugReporter.reportBug(new BugInstance(this, getMethodName().equals("setUp") ? "IJU_SETUP_NO_SUPER"
198                             : "IJU_TEARDOWN_NO_SUPER", NORMAL_PRIORITY).addClassAndMethod(this).addMethod(we, superMethod)
199                             .describe(MethodAnnotation.METHOD_OVERRIDDEN).addSourceLine(this, offset));
200                 }
201             }
202         }
203     }
204 
205     @Override
206     public void sawOpcode(int seen) {
207         switch (state) {
208         case SEEN_NOTHING:
209             if (seen == Const.ALOAD_0) {
210                 state = SEEN_ALOAD_0;
211             }
212             break;
213 
214         case SEEN_ALOAD_0:
215             if ((seen == Const.INVOKESPECIAL) && (getNameConstantOperand().equals(getMethodName()))
216                     && (getSigConstantOperand().equals("()V"))) {
217                 sawSuperCall = true;
218             }
219             state = SEEN_NOTHING;
220             break;
221         default:
222             state = SEEN_NOTHING;
223         }
224     }
